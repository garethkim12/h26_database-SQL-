\-- 부서 테이블 생성

&#x20;   empno NUMBER PRIMARY KEY,

&#x20;   address VARCHAR2(100),

&#x20;   sex VARCHAR2(10),

&#x20;   position VARCHAR2(20),

&#x20;   depno NUMBER REFERENCES Department(depno)

\-- 프로젝트 테이블 생성

&#x20;   projno NUMBER PRIMARY KEY,
